\begin{frame}{Abstract}
  \begin{quote}
    We examine contemporary naming practices using a combination of qualitative
    and quantitative data: parental naming stories from the Baby Calendar
    website, historical and modern name guides, and large-scale frequency data
    from Heisei Namae Jiten and Meiji Yasuda.

    Our analysis shows a clear shift: traditional gender distinctions are
    becoming less rigid, with more gender-neutral names and a softening of
    masculine naming patterns.

    Placing Japan in a broader perspective, we argue that these developments
    reflect a wider global trend in the rethinking of gender and identity
    through personal names.
  \end{quote}
\end{frame}

\section{Datasets and Coverage}

\begin{frame}{Three Complementary Datasets}
  \begin{description}
    \item[\textbf{Baby Calendar (BC)}] Online parenting site; names +
      pronunciations + parental commentary, 2008--2022.
      \(\approx\)16,000 name entries per year.
    \item[\textbf{Heisei Namae Jiten (HS)}] Commercial name dictionary;
      frequency counts by orthography, 1989--2009.
      \(\approx\)300,000--600,000 records/year.
    \item[\textbf{Meiji Yasuda (MY)}] Insurance-company survey; top-100 baby
      names per year, 2004--2024.
  \end{description}
\end{frame}

\section{Irregular Readings}

\begin{frame}{What Is an Irregular Reading?}
  \begin{itemize}
    \item Japanese names can be read in ways not predicted by standard
      phonological rules (\emph{nanori})
    \item \ja{心} \textit{kokoro} `heart' → read as \textit{noa} in a name
    \item Irregular readings signal creativity — and sometimes controversy
    \item \emph{Kira-kira} `sparkle' names
  \end{itemize}
\end{frame}

\section{Gender and Names}

\begin{frame}{Gender Coding of Names}
  \begin{itemize}
    \item We define an \textbf{F-ratio} = female count / (male + female)
    \item F-ratio = 0 → exclusively male; F-ratio = 1 → exclusively female
    \item Gender-neutral names: F-ratio $\in [0.2, 0.8]$
  \end{itemize}
\end{frame}

\section{Japan in Global Perspective}

\begin{frame}{A Global Trend}
  \begin{itemize}
    \item Gender-neutral naming is rising across many societies
      \begin{itemize}
        \item English-speaking world: \textit{River}, \textit{Sage}, \textit{Quinn}
        \item Scandinavia: explicit de-gendering in law and culture
        \item South Korea, China: parallel trends in East Asia
      \end{itemize}
    \item Drivers: changing gender norms, social media visibility,
      desire for individuality
    \item Japan: slower but measurable shift — constrained by kanji
      gender-marking conventions and registry rules
  \end{itemize}
\end{frame}

\begin{frame}{What Constrains Change in Japan?}
  \begin{itemize}
    \item Kanji carry gender associations (\ja{美}/\ja{子}/\ja{花} = feminine;
      \ja{郎}/\ja{夫}/\ja{男} = masculine)
    \item Registry law restricts name characters to approved list
    \item Cultural pressure: names must be readable, not embarrassing
    \item But: \emph{pronunciation} is more flexible than orthography —
      gender-neutral sounds spread faster than gender-neutral kanji
  \end{itemize}
\end{frame}

\begin{frame}{Parental Voice: Baby Calendar Naming Stories}
  \begin{itemize}
    \item ``We wanted a name that works for any gender''
    \item ``The sound should be cute but not too girly''
    \item ``We chose \ja{蓮} because it sounds natural in both Japanese and
      internationally''
    \item Themes: nature, universality, softness — regardless of child's gender
  \end{itemize}
  \vspace{1em}
  \begin{block}{Observation}
    Parental narratives show conscious gender-neutralising intent,
    even when the name itself remains statistically gendered.
  \end{block}
\end{frame}

\section{Conclusion}

\begin{frame}{Summary}
  \begin{enumerate}
    \item \textbf{Diversity} is increasing: more names, fewer shared top-names,
      rising irregular readings — all trends point toward greater individuality
    \item \textbf{Gender coding} remains strong but is softening: middle bins
      of the F-ratio distribution are growing; gender-shared names rising
    \item \textbf{Kanji position} reveals cultural fashion cycles:
      specific characters rise and fall in specific positions over decades
    \item \textbf{Global context}: Japan's trajectory parallels international
      trends toward more fluid gender expression through naming,
      moderated by linguistic and legal constraints
  \end{enumerate}
\end{frame}

\begin{frame}{Future Work}
  \begin{itemize}
    \item Deeper qualitative analysis of naming stories (Baby Calendar)
    \item Cross-linguistic comparison with Korean and Chinese data
    \item Modelling the spread of new name fashions as diffusion processes
    \item Impact of the 2024 Family Register Law revision on name choice
  \end{itemize}
\end{frame}
